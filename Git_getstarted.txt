Befehl:								Auswirkung:

# Allgemein
cd									change directory
ls									inhalt des directory anzeigen
git add .							alle dateien stagen
git commit							alle dateien die gestaged sind erhöhen
git add <name>						nur diese datei stagen
git status							status der dateien und auf version mit online repo kontrollieren


# git einrichten
git init							aktuelles directory aktivieren und als main definieren
git remote -v						Upload URL's anzeigen lassen
git remote set-url					Upload URL's ändern (für SSH-Key)
git config --global user.name		Benutzernamen ändern
git config --global user.mail		Mailadresse ändern
git remote rm						löschen von remote url
